# Loans + Admin Loan Portal Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the client-facing loan views (loan list, loan details with installments, loan application form) and the employee-facing loan management portal (loan requests approval/rejection, all loans view with filters).

**Architecture:** Client pages: LoanListPage (all loans sorted by amount), LoanDetailsPage (loan details + installment history), LoanApplicationPage (multi-step loan application form). Employee pages: AdminLoanRequestsPage (pending requests with approve/reject), AdminLoansPage (all loans with filters). Loan application uses Redux Toolkit for multi-step form flow. Server data via TanStack Query.

**Tech Stack:** React 19, TypeScript, Shadcn UI, TanStack Query, Redux Toolkit, React Hook Form + Zod, Jest + RTL

**Depends on:** Plan 1 (Foundation — Loan, LoanRequest types, loan API stubs, constants)

---

## Assumptions

1. `getLoans()` returns client's loans sorted by amount descending (server-side).
2. `getLoan(id)` returns loan details including installments array.
3. `createLoanRequest(payload)` submits a new loan application. Backend validates and returns the request.
4. `getLoanRequests(filters)` (employee) returns all loan requests with filtering by type, account number.
5. `approveLoanRequest(id)` and `rejectLoanRequest(id)` update the request status.
6. `getAllLoans(filters)` (employee) returns all loans with filtering by type, account, status.
7. Interest rate calculation is server-side. Frontend displays the rates from the loan entity.
8. Installment type (Rata): `id`, `loan_id`, `amount`, `interest_rate`, `currency`, `expected_date`, `actual_date`, `status` (PAID, UNPAID, LATE).

// src/hooks/useLoans.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import { getLoans, getLoan, createLoanRequest } from '@/lib/api/loans'
import type { CreateLoanRequest } from '@/types/loan'

export function useLoans() {
  return useQuery({
    queryKey: ['loans'],
    queryFn: getLoans,
  })
}

export function useLoanById(id: number) {
  return useQuery({
    queryKey: ['loans', id],
    queryFn: () => getLoan(id),
    enabled: !!id,
  })
}

export function useCreateLoanRequest() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (payload: CreateLoanRequest) => createLoanRequest(payload),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['loans'] })
    },
  })
}
```

// src/hooks/useAdminLoans.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import {
  getLoanRequests,
  approveLoanRequest,
  rejectLoanRequest,
  getAllLoans,
} from '@/lib/api/loans'
import type { LoanFilters } from '@/types/loan'

interface LoanRequestFilters {
  loan_type?: string
  account_number?: string
}

export function useLoanRequests(filters: LoanRequestFilters) {
  return useQuery({
    queryKey: ['loan-requests', filters],
    queryFn: () => getLoanRequests(filters),
  })
}

export function useApproveLoanRequest() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (id: number) => approveLoanRequest(id),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['loan-requests'] })
      queryClient.invalidateQueries({ queryKey: ['all-loans'] })
    },
  })
}

export function useRejectLoanRequest() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (id: number) => rejectLoanRequest(id),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['loan-requests'] })
    },
  })
}

export function useAllLoans(filters: LoanFilters) {
  return useQuery({
    queryKey: ['all-loans', filters],
    queryFn: () => getAllLoans(filters),
  })
}
```

// src/store/slices/loanApplicationSlice.ts
import { createSlice, type PayloadAction } from '@reduxjs/toolkit'

export interface LoanApplicationData {
  loan_type: string
  interest_type: string
  amount: number
  currency: string
  purpose: string
  monthly_income: number
  employment_status: string
  employment_period: number
  installment_count: number
  phone: string
  account_number: string
}

export interface LoanApplicationState {
  step: 'form' | 'review' | 'success'
  data: LoanApplicationData | null
  submitting: boolean
  error: string | null
}

const initialState: LoanApplicationState = {
  step: 'form',
  data: null,
  submitting: false,
  error: null,
}

const loanApplicationSlice = createSlice({
  name: 'loanApplication',
  initialState,
  reducers: {
    setLoanApplicationStep(state, action: PayloadAction<LoanApplicationState['step']>) {
      state.step = action.payload
    },
    setLoanApplicationData(state, action: PayloadAction<LoanApplicationData>) {
      state.data = action.payload
    },
    resetLoanApplication() {
      return initialState
    },
    setLoanApplicationSubmitting(state, action: PayloadAction<boolean>) {
      state.submitting = action.payload
    },
    setLoanApplicationError(state, action: PayloadAction<string | null>) {
      state.error = action.payload
    },
  },
})

export const {
  setLoanApplicationStep,
  setLoanApplicationData,
  resetLoanApplication,
  setLoanApplicationSubmitting,
  setLoanApplicationError,
} = loanApplicationSlice.actions
export default loanApplicationSlice.reducer
```

// Add to src/lib/utils/validation.ts

const LOAN_TYPES = ['CASH', 'HOUSING', 'AUTO', 'REFINANCING', 'STUDENT'] as const
const INTEREST_TYPES = ['FIXED', 'VARIABLE'] as const
const EMPLOYMENT_STATUSES = ['PERMANENT', 'TEMPORARY', 'UNEMPLOYED'] as const
const STANDARD_INSTALLMENTS = [12, 24, 36, 48, 60, 72, 84] as const
const HOUSING_INSTALLMENTS = [60, 120, 180, 240, 300, 360] as const

export const loanApplicationSchema = z.object({
  loan_type: z.enum(LOAN_TYPES),
  interest_type: z.enum(INTEREST_TYPES),
  amount: z.number().positive('Amount must be greater than 0'),
  currency: z.string().min(1),
  purpose: z.string().min(1, 'Purpose is required'),
  monthly_income: z.number().positive('Monthly income must be greater than 0'),
  employment_status: z.enum(EMPLOYMENT_STATUSES),
  employment_period: z.number().min(0, 'Employment period must be non-negative'),
  installment_count: z.number(),
  phone: z.string().min(1, 'Phone number is required'),
  account_number: z.string().regex(/^\d{18}$/, 'Account number must be 18 digits'),
}).superRefine((data, ctx) => {
  const validCounts = data.loan_type === 'HOUSING'
    ? HOUSING_INSTALLMENTS as readonly number[]
    : STANDARD_INSTALLMENTS as readonly number[]

  if (!validCounts.includes(data.installment_count)) {
    ctx.addIssue({
      code: z.ZodIssueCode.custom,
      message: `Invalid installment count for ${data.loan_type} loan`,
      path: ['installment_count'],
    })
  }
})
```

// src/pages/LoanApplicationPage.tsx
import { useEffect } from 'react'
import { useAppDispatch, useAppSelector } from '@/hooks/useAppDispatch'
import { useClientAccounts } from '@/hooks/useAccounts'
import { useCreateLoanRequest } from '@/hooks/useLoans'
import {
  setLoanApplicationStep,
  setLoanApplicationData,
  resetLoanApplication,
} from '@/store/slices/loanApplicationSlice'
import { LoanApplicationForm } from '@/components/loans/LoanApplicationForm'
import { Button } from '@/components/ui/button'
import { useNavigate } from 'react-router-dom'
import type { LoanApplicationData } from '@/store/slices/loanApplicationSlice'

export function LoanApplicationPage() {
  const dispatch = useAppDispatch()
  const navigate = useNavigate()
  const { step, data } = useAppSelector((s) => s.loanApplication)
  const { data: accounts, isLoading } = useClientAccounts()
  const createLoan = useCreateLoanRequest()

  useEffect(() => {
    return () => { dispatch(resetLoanApplication()) }
  }, [dispatch])

  if (isLoading) return <p>Učitavanje...</p>

  const handleSubmit = (formData: LoanApplicationData) => {
    dispatch(setLoanApplicationData(formData))
    // Submit directly (or add review step)
    createLoan.mutate(formData as any, {
      onSuccess: () => dispatch(setLoanApplicationStep('success')),
    })
  }

  if (step === 'success') {
    return (
      <div className="space-y-4 text-center">
        <h2 className="text-xl font-semibold">Zahtev za kredit uspešno podnet!</h2>
        <Button onClick={() => navigate('/loans')}>Nazad na kredite</Button>
      </div>
    )
  }

  return (
    <div className="space-y-4">
      <Button variant="ghost" onClick={() => navigate('/loans')}>← Nazad</Button>
      <LoanApplicationForm accounts={accounts ?? []} onSubmit={handleSubmit} />
    </div>
  )
}
