# Payments Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the complete payments subsystem: new payment (between different clients), internal transfer (between own accounts, same currency), payment recipients CRUD, payment history with filters, transaction details with client-side PDF receipt.

**Architecture:** Four client-only pages behind `/payments/*` routes. New Payment and Internal Transfer use a 2-step flow (form → confirmation) managed via Redux Toolkit slice. Payment recipients is a standalone CRUD page. Payment history lists all transactions with filter/sort + detail view with PDF download. Server data via TanStack Query; multi-step payment/transfer submission via Redux `createAsyncThunk`.

**Tech Stack:** React 19, TypeScript, Shadcn UI, TanStack Query, Redux Toolkit, React Hook Form + Zod, jsPDF (client-side PDF), Jest + RTL

**Depends on:** Plan 1 (Foundation — types, API stubs, constants), Plan 2 (useAccounts hooks), Plan 3 (usePayments query hook, useExchange hooks)

---

## Assumptions

1. `createPayment` API returns the created Payment object with server-generated `id` and `order_number`.
2. `createInternalTransfer` API handles same-currency transfers between own accounts. Returns a Payment-like object.
3. Payment recipients are per-client — the API scopes them to the authenticated user.
4. PDF receipt is generated client-side using jsPDF (no backend PDF endpoint).
5. Payment code list (šifre plaćanja) is available from `src/constants/payment-codes.ts` (Plan 1).
6. Verification step (mobile app) is skipped per architectural decision.
7. The `usePayments` hook from Plan 3 provides `usePayments(filters)` query. This plan adds mutation hooks.

```typescript
// src/store/slices/paymentSlice.ts
import { createSlice, createAsyncThunk, type PayloadAction } from '@reduxjs/toolkit'
import { createPayment } from '@/lib/api/payments'
import type { Payment, CreatePaymentRequest } from '@/types/payment'

export interface PaymentFormData {
  recipient_name: string
  recipient_account: string
  amount: number
  payment_code: string
  reference_number?: string
  purpose: string
  from_account: string
  currency?: string
}

export interface PaymentFlowState {
  step: 'form' | 'confirmation' | 'success'
  formData: PaymentFormData | null
  submitting: boolean
  error: string | null
  result: Payment | null
}

const initialState: PaymentFlowState = {
  step: 'form',
  formData: null,
  submitting: false,
  error: null,
  result: null,
}

export const submitPayment = createAsyncThunk(
  'payment/submit',
  async (payload: CreatePaymentRequest, { rejectWithValue }) => {
    try {
      return await createPayment(payload)
    } catch (err: unknown) {
      const error = err as { response?: { data?: { message?: string } } }
      return rejectWithValue(error.response?.data?.message ?? 'Payment failed')
    }
  }
)

const paymentSlice = createSlice({
  name: 'payment',
  initialState,
  reducers: {
    setPaymentStep(state, action: PayloadAction<PaymentFlowState['step']>) {
      state.step = action.payload
    },
    setPaymentFormData(state, action: PayloadAction<PaymentFormData>) {
      state.formData = action.payload
    },
    resetPaymentFlow() {
      return initialState
    },
  },
  extraReducers: (builder) => {
    builder
      .addCase(submitPayment.pending, (state) => {
        state.submitting = true
        state.error = null
      })
      .addCase(submitPayment.fulfilled, (state, action) => {
        state.submitting = false
        state.result = action.payload
        state.step = 'success'
      })
      .addCase(submitPayment.rejected, (state, action) => {
        state.submitting = false
        state.error = action.payload as string
      })
  },
})

export const { setPaymentStep, setPaymentFormData, resetPaymentFlow } = paymentSlice.actions
export default paymentSlice.reducer
```

```typescript
// Add to src/hooks/usePayments.ts
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query'
import {
  createPayment,
  createInternalTransfer,
  getPayment,
} from '@/lib/api/payments'
import type { CreatePaymentRequest, Payment } from '@/types/payment'

export function useCreatePayment() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (payload: CreatePaymentRequest) => createPayment(payload),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['payments'] })
      queryClient.invalidateQueries({ queryKey: ['accounts'] })
    },
  })
}

export function useCreateInternalTransfer() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (payload: { from_account: string; to_account: string; amount: number }) =>
      createInternalTransfer(payload),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['payments'] })
      queryClient.invalidateQueries({ queryKey: ['accounts'] })
    },
  })
}

export function usePaymentById(id: number) {
  return useQuery({
    queryKey: ['payments', id],
    queryFn: () => getPayment(id),
    enabled: !!id,
  })
}
```

```typescript
// src/hooks/usePaymentRecipients.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import {
  getPaymentRecipients,
  createPaymentRecipient,
  updatePaymentRecipient,
  deletePaymentRecipient,
} from '@/lib/api/payments'
import type { PaymentRecipient } from '@/types/payment-recipient'

export function usePaymentRecipients() {
  return useQuery({
    queryKey: ['payment-recipients'],
    queryFn: getPaymentRecipients,
  })
}

export function useCreateRecipient() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (data: { name: string; account_number: string }) =>
      createPaymentRecipient(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['payment-recipients'] })
    },
  })
}

export function useUpdateRecipient() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: ({ id, ...data }: { id: number; name: string; account_number: string }) =>
      updatePaymentRecipient(id, data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['payment-recipients'] })
    },
  })
}

export function useDeleteRecipient() {
  const queryClient = useQueryClient()
  return useMutation({
    mutationFn: (id: number) => deletePaymentRecipient(id),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['payment-recipients'] })
    },
  })
}
```

```typescript
// Add to src/lib/utils/validation.ts

export const newPaymentSchema = z.object({
  recipient_name: z.string().min(1, 'Recipient name is required'),
  recipient_account: z.string().regex(/^\d{18}$/, 'Account number must be 18 digits'),
  amount: z.number().positive('Amount must be greater than 0'),
  payment_code: z.string().regex(/^2\d{2}$/, 'Payment code must be 3 digits starting with 2'),
  reference_number: z.string().optional(),
  purpose: z.string().min(1, 'Purpose is required'),
  from_account: z.string().regex(/^\d{18}$/, 'Account number must be 18 digits'),
}).refine(
  (data) => data.from_account !== data.recipient_account,
  { message: 'Sender and recipient accounts must be different', path: ['recipient_account'] }
)

export const internalTransferSchema = z.object({
  from_account: z.string().regex(/^\d{18}$/, 'Account number must be 18 digits'),
  to_account: z.string().regex(/^\d{18}$/, 'Account number must be 18 digits'),
  amount: z.number().positive('Amount must be greater than 0'),
}).refine(
  (data) => data.from_account !== data.to_account,
  { message: 'Source and destination accounts must be different', path: ['to_account'] }
)

export const recipientSchema = z.object({
  name: z.string().min(1, 'Name is required'),
  account_number: z.string().regex(/^\d{18}$/, 'Account number must be 18 digits'),
})
```

```typescript
// src/lib/utils/receipt-pdf.ts
import jsPDF from 'jspdf'
import type { Payment } from '@/types/payment'
import { formatCurrency, formatAccountNumber } from '@/lib/utils/format'

export function generateReceiptPdf(payment: Payment): void {
  const doc = new jsPDF()
  const pageWidth = doc.internal.pageSize.getWidth()
  const centerX = pageWidth / 2
  let y = 20

  // Header
  doc.setFontSize(18)
  doc.text('Potvrda o plaćanju', centerX, y, { align: 'center' })
  y += 15

  doc.setFontSize(10)
  doc.line(20, y, pageWidth - 20, y)
  y += 10

  // Details
  doc.setFontSize(11)
  const details = [
    ['Broj naloga:', payment.order_number],
    ['Sa računa:', formatAccountNumber(payment.from_account)],
    ['Na račun:', formatAccountNumber(payment.to_account)],
    ['Primalac:', payment.recipient_name],
    ['Iznos:', formatCurrency(payment.amount, payment.currency)],
    ['Krajnji iznos:', formatCurrency(payment.final_amount, payment.currency)],
    ['Provizija:', formatCurrency(payment.commission, payment.currency)],
    ['Šifra plaćanja:', payment.payment_code],
    ['Svrha:', payment.purpose],
    ['Status:', payment.status],
    ['Datum:', new Date(payment.timestamp).toLocaleDateString('sr-Latn-RS')],
  ]

  for (const [label, value] of details) {
    doc.text(label, 25, y)
    doc.text(value, 80, y)
    y += 8
  }

  y += 5
  doc.line(20, y, pageWidth - 20, y)

  doc.save(`receipt-${payment.order_number}.pdf`)
}
```

```typescript
// src/pages/NewPaymentPage.tsx
import { useState } from 'react'
import { useAppDispatch, useAppSelector } from '@/hooks/useAppDispatch'
import { useClientAccounts } from '@/hooks/useAccounts'
import { usePaymentRecipients, useCreatePaymentRecipient } from '@/hooks/usePaymentRecipients'
import {
  setPaymentStep,
  setPaymentFormData,
  resetPaymentFlow,
  submitPayment,
} from '@/store/slices/paymentSlice'
import { NewPaymentForm } from '@/components/payments/NewPaymentForm'
import { PaymentConfirmation } from '@/components/payments/PaymentConfirmation'
import { Button } from '@/components/ui/button'
import { useNavigate } from 'react-router-dom'
import { useEffect } from 'react'
import type { PaymentFormData } from '@/store/slices/paymentSlice'

export function NewPaymentPage() {
  const dispatch = useAppDispatch()
  const navigate = useNavigate()
  const { step, formData, submitting, result } = useAppSelector((s) => s.payment)
  const { data: accounts, isLoading: loadingAccounts } = useClientAccounts()
  const { data: recipients, isLoading: loadingRecipients } = usePaymentRecipients()

  useEffect(() => {
    return () => { dispatch(resetPaymentFlow()) }
  }, [dispatch])

  if (loadingAccounts || loadingRecipients) {
    return <p>Učitavanje...</p>
  }

  const handleFormSubmit = (data: PaymentFormData) => {
    dispatch(setPaymentFormData(data))
    dispatch(setPaymentStep('confirmation'))
  }

  const handleConfirm = () => {
    if (!formData) return
    dispatch(submitPayment({
      from_account: formData.from_account,
      to_account: formData.recipient_account,
      amount: formData.amount,
      recipient_name: formData.recipient_name,
      payment_code: formData.payment_code,
      reference_number: formData.reference_number,
      purpose: formData.purpose,
    }))
  }

  const handleBack = () => dispatch(setPaymentStep('form'))

  const createRecipient = useCreatePaymentRecipient()
  const [recipientSaved, setRecipientSaved] = useState(false)

  // Check if the payment recipient is already saved
  const isNewRecipient = result && formData
    ? !(recipients ?? []).some((r) => r.account_number === formData.to_account)
    : false

  if (step === 'success' && result) {
    return (
      <div className="space-y-4 text-center">
        <h2 className="text-xl font-semibold">Plaćanje uspešno!</h2>
        <p>Broj naloga: {result.order_number}</p>
        <div className="flex justify-center gap-3">
          <Button onClick={() => navigate(`/payments/history/${result.id}`)}>Detalji</Button>
          <Button variant="outline" onClick={() => dispatch(resetPaymentFlow())}>Novo plaćanje</Button>
        </div>
        {isNewRecipient && !recipientSaved && formData && (
          <Button
            variant="outline"
            onClick={() => {
              createRecipient.mutate(
                { name: formData.recipient_name, account_number: formData.to_account },
                { onSuccess: () => setRecipientSaved(true) }
              )
            }}
          >
            Dodaj primaoca
          </Button>
        )}
        {recipientSaved && <p className="text-sm text-muted-foreground">Primalac sačuvan.</p>}
      </div>
    )
  }

  if (step === 'confirmation' && formData) {
    return <PaymentConfirmation data={formData} onConfirm={handleConfirm} onBack={handleBack} submitting={submitting} />
  }

  return (
    <NewPaymentForm
      accounts={accounts ?? []}
      recipients={recipients ?? []}
      onSubmit={handleFormSubmit}
    />
  )
}
```

```typescript
import { useState } from 'react'
import { useClientAccounts } from '@/hooks/useAccounts'
import { useCreateInternalTransfer } from '@/hooks/usePayments'
import { InternalTransferForm } from '@/components/payments/InternalTransferForm'
import { TransferConfirmation } from '@/components/payments/TransferConfirmation'
import { Button } from '@/components/ui/button'
import { useNavigate } from 'react-router-dom'

interface TransferData {
  from_account: string
  to_account: string
  amount: number
}

export function PaymentTransferPage() {
  const navigate = useNavigate()
  const { data: accounts, isLoading } = useClientAccounts()
  const createTransfer = useCreateInternalTransfer()
  const [step, setStep] = useState<'form' | 'confirmation' | 'success'>('form')
  const [formData, setFormData] = useState<TransferData | null>(null)

  if (isLoading) return <p>Učitavanje...</p>

  const handleSubmitForm = (data: TransferData) => {
    setFormData(data)
    setStep('confirmation')
  }

  const handleConfirm = () => {
    if (!formData) return
    createTransfer.mutate(formData, {
      onSuccess: () => setStep('success'),
    })
  }

  if (step === 'success') {
    return (
      <div className="space-y-4 text-center">
        <h2 className="text-xl font-semibold">Prenos uspešan!</h2>
        <Button onClick={() => { setStep('form'); setFormData(null) }}>Novi prenos</Button>
      </div>
    )
  }

  if (step === 'confirmation' && formData) {
    const acc = accounts?.find((a) => a.account_number === formData.from_account)
    return (
      <TransferConfirmation
        fromAccount={formData.from_account}
        toAccount={formData.to_account}
        amount={formData.amount}
        currency={acc?.currency ?? 'RSD'}
        onConfirm={handleConfirm}
        onBack={() => setStep('form')}
        submitting={createTransfer.isPending}
      />
    )
  }

  return <InternalTransferForm accounts={accounts ?? []} onSubmit={handleSubmitForm} />
}
```

```typescript
import { useParams, useNavigate } from 'react-router-dom'
import { usePaymentById } from '@/hooks/usePayments'
import { TransactionDetails } from '@/components/payments/TransactionDetails'
import { Button } from '@/components/ui/button'

export function PaymentDetailsPage() {
  const { id } = useParams<{ id: string }>()
  const navigate = useNavigate()
  const { data: payment, isLoading, error } = usePaymentById(Number(id))

  if (isLoading) return <p>Učitavanje...</p>
  if (error || !payment) return <p>Transakcija nije pronađena.</p>

  return (
    <div className="space-y-4">
      <Button variant="ghost" onClick={() => navigate(-1)}>← Nazad</Button>
      <TransactionDetails payment={payment} />
    </div>
  )
}
```
